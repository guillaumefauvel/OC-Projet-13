319 class OracleParam:
320 """
321 Wrapper object for formatting parameters for Oracle. If the string
322 representation of the value is large enough (greater than 4000 characters)
323 the input size needs to be set as CLOB. Alternatively, if the parameter
324 has an `input_size` attribute, then the value of the `input_size` attribute
325 will be used instead. Otherwise, no input size will be set for the
326 parameter when executing the query.
327 """

329 def __init__(self, param, cursor, strings_only=False):
330 # With raw SQL queries, datetimes can reach this function
331 # without being converted by DateTimeField.get_db_prep_value.
332 if settings.USE_TZ and (isinstance(param, datetime.datetime) and
333 not isinstance(param, Oracle_datetime)):
334 param = Oracle_datetime.from_datetime(param)
335  
336 string_size = 0
337 # Oracle doesn't recognize True and False correctly.
338 if param is True:
339 param = 1
340 elif param is False:
341 param = 0
342 if hasattr(param, 'bind_parameter'):
343 self.force_bytes = param.bind_parameter(cursor)
344 elif isinstance(param, (Database.Binary, datetime.timedelta)):
345 self.force_bytes = param
346 else:
347 # To transmit to the database, we need Unicode if supported
348 # To get size right, we must consider bytes.
349 self.force_bytes = force_str(param, cursor.charset, strings_only)
350 if isinstance(self.force_bytes, str):
351 # We could optimize by only converting up to 4000 bytes here
352 string_size = len(force_bytes(param, cursor.charset, strings_only))
353 if hasattr(param, 'input_size'):
354 # If parameter has `input_size` attribute, use that.
355 self.input_size = param.input_size
356 elif string_size > 4000:
357 # Mark any string param greater than 4000 characters as a CLOB.
358 self.input_size = Database.CLOB
359 elif isinstance(param, datetime.datetime):
360 self.input_size = Database.TIMESTAMP
361 else:
362 self.input_size = None

479 def _fix_for_params(self, query, params, unify_by_values=False):
480 # cx_Oracle wants no trailing ';' for SQL statements. For PL/SQL, it
481 # it does want a trailing ';' but not a trailing '/'. However, these
482 # characters must be included in the original query in case the query
483 # is being passed to SQL*Plus.
484 if query.endswith(';') or query.endswith('/'):
485 query = query[:-1]
486 if params is None:
487 params = []
488 elif hasattr(params, 'keys'):
489 # Handle params as dict
490 args = {k: ":%s" % k for k in params}
491 query = query % args
492 elif unify_by_values and params:
493 # Handle params as a dict with unified query parameters by their
494 # values. It can be used only in single query execute() because
495 # executemany() shares the formatted query with each of the params
496 # list. e.g. for input params = [0.75, 2, 0.75, 'sth', 0.75]
497 # params_dict = {0.75: ':arg0', 2: ':arg1', 'sth': ':arg2'}
498 # args = [':arg0', ':arg1', ':arg0', ':arg2', ':arg0']
499 # params = {':arg0': 0.75, ':arg1': 2, ':arg2': 'sth'}
500 params_dict = {param: ':arg%d' % i for i, param in enumerate(set(params))}
501 args = [params_dict[param] for param in params]
502 params = {value: key for key, value in params_dict.items()}
503 query = query % tuple(args)
504 else:
505 # Handle params as sequence
506 args = [(':arg%d' % i) for i in range(len(params))]
507 query = query % tuple(args)
508 return query, self._format_params(params)

510 def execute(self, query, params=None):
511 query, params = self._fix_for_params(query, params, unify_by_values=True)
512 self._guess_input_sizes([params])
513 with wrap_oracle_errors():
514 return self.cursor.execute(query, self._param_generator(params))

516 def executemany(self, query, params=None):
517 if not params:
518 # No params given, nothing to do
519 return None
520 # uniform treatment for sequences and iterables
521 params_iter = iter(params)
522 query, firstparams = self._fix_for_params(query, next(params_iter))
523 # we build a list of formatted params; as we're going to traverse it
524 # more than once, we can't make it lazy by using a generator
525 formatted = [firstparams] + [self._format_params(p) for p in params_iter]
526 self._guess_input_sizes(formatted)
527 with wrap_oracle_errors():
528 return self.cursor.executemany(query, [self._param_generator(p) for p in formatted])
