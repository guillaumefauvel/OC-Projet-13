47 class Color:
48 """
49 Color conversion support class
50  
51 Example:
52  
53 ::
54  
55 from selenium.webdriver.support.color import Color
56  
57 print(Color.from_string('#00ff33').rgba)
58 print(Color.from_string('rgb(1, 255, 3)').hex)
59 print(Color.from_string('blue').rgba)
60 """
61  
62 @classmethod
63 def from_string(cls, str_: str) -> Color:
64 import re
65  
66 class Matcher:
67 match_obj: Optional[Match[str]]
68  
69 def __init__(self) -> None:
70 self.match_obj = None
71  
72 def match(self, pattern: str, str_: str) -> Optional[Match[str]]:
73 self.match_obj = re.match(pattern, str_)
74 return self.match_obj
75  
76 @property
77 def groups(self) -> Sequence[str]:
78 return () if not self.match_obj else self.match_obj.groups()
79  
80 m = Matcher()
81  
82 if m.match(RGB_PATTERN, str_):
83 return cls(*m.groups)
84 elif m.match(RGB_PCT_PATTERN, str_):
85 rgb = tuple([float(each) / 100 * 255 for each in m.groups])
86 return cls(*rgb)
87 elif m.match(RGBA_PATTERN, str_):
88 return cls(*m.groups)
89 elif m.match(RGBA_PCT_PATTERN, str_):
90 rgba = tuple(
91 [float(each) / 100 * 255 for each in m.groups[:3]] + [m.groups[3]]) # type: ignore
92 return cls(*rgba)
93 elif m.match(HEX_PATTERN, str_):
94 rgb = tuple([int(each, 16) for each in m.groups])
95 return cls(*rgb)
96 elif m.match(HEX3_PATTERN, str_):
97 rgb = tuple([int(each * 2, 16) for each in m.groups])
98 return cls(*rgb)
99 elif m.match(HSL_PATTERN, str_) or m.match(HSLA_PATTERN, str_):
100 return cls._from_hsl(*m.groups)
101 elif str_.upper() in Colors.keys():
102 return Colors[str_.upper()]
103 else:
104 raise ValueError("Could not convert %s into color" % str_)

106 @classmethod
107 def _from_hsl(cls, h: ParseableFloat, s: ParseableFloat, light: ParseableFloat,
108 a: ParseableFloat = 1) -> Color:
109 h = float(h) / 360
110 s = float(s) / 100
111 _l = float(light) / 100
112  
113 if s == 0:
114 r = _l
115 g = r
116 b = r
117 else:
118 luminocity2 = _l * (1 + s) if _l < 0.5 else _l + s - _l * s
119 luminocity1 = 2 * _l - luminocity2
120  
121 def hue_to_rgb(lum1: float, lum2: float, hue: float) -> float:
122 if hue < 0.0:
123 hue += 1
124 if hue > 1.0:
125 hue -= 1
126  
127 if hue < 1.0 / 6.0:
128 return lum1 + (lum2 - lum1) * 6.0 * hue
129 elif hue < 1.0 / 2.0:
130 return lum2
131 elif hue < 2.0 / 3.0:
132 return lum1 + (lum2 - lum1) * ((2.0 / 3.0) - hue) * 6.0
133 else:
134 return lum1
135  
136 r = hue_to_rgb(luminocity1, luminocity2, h + 1.0 / 3.0)
137 g = hue_to_rgb(luminocity1, luminocity2, h)
138 b = hue_to_rgb(luminocity1, luminocity2, h - 1.0 / 3.0)
139  
140 return cls(round(r * 255), round(g * 255), round(b * 255), a)
141  
142 def __init__(self, red: ParseableInt, green: ParseableInt, blue: ParseableInt,
143 alpha: ParseableFloat = 1) -> None:
144 self.red = int(red)
145 self.green = int(green)
146 self.blue = int(blue)
147 self.alpha = "1" if float(alpha) == 1 else str(float(alpha) or 0)
148  
149 @property
150 def rgb(self) -> str:
151 return "rgb(%d, %d, %d)" % (self.red, self.green, self.blue)
152  
153 @property
154 def rgba(self) -> str:
155 return "rgba(%d, %d, %d, %s)" % (self.red, self.green, self.blue, self.alpha)
156  
157 @property
158 def hex(self) -> str:
159 return "#%02x%02x%02x" % (self.red, self.green, self.blue)
160  
161 def __eq__(self, other: object) -> bool:
162 if isinstance(other, Color):
163 return self.rgba == other.rgba
164 return NotImplemented
165  
166 def __ne__(self, other: Any) -> bool:
167 result = self.__eq__(other)
168 if result is NotImplemented:
169 return result
170 return not result
171  
172 def __hash__(self) -> int:
173 return hash((self.red, self.green, self.blue, self.alpha))
174  
175 def __repr__(self) -> str:
176 return "Color(red=%d, green=%d, blue=%d, alpha=%s)" % (
177 self.red, self.green, self.blue, self.alpha)
178  
179 def __str__(self) -> str:
180 return "Color: %s" % self.rgba
