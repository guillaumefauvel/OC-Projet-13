42 def none_guard(func):
43 """
44 Decorator that returns None if any of the arguments to the decorated
45 function are None. Many SQL functions return NULL if any of their arguments
46 are NULL. This decorator simplifies the implementation of this for the
47 custom functions registered below.
48 """
49 @functools.wraps(func)
50 def wrapper(*args, **kwargs):
51 return None if None in args else func(*args, **kwargs)
52 return wrapper
