23 def check_token(self, user, token):
24 """
25 Check that a password reset token is correct for a given user.
26 """
27 if not (user and token):
28 return False
29 # Parse the token
30 try:
31 ts_b36, _ = token.split("-")
32 except ValueError:
33 return False
34  
35 try:
36 ts = base36_to_int(ts_b36)
37 except ValueError:
38 return False
39  
40 # Check that the timestamp/uid has not been tampered with
41 if not constant_time_compare(self._make_token_with_timestamp(user, ts), token):
42 return False
43  
44 # Check the timestamp is within limit. Timestamps are rounded to
45 # midnight (server time) providing a resolution of only 1 day. If a
46 # link is generated 5 minutes before midnight and used 6 minutes later,
47 # that counts as 1 day. Therefore, PASSWORD_RESET_TIMEOUT_DAYS = 1 means
48 # "at least 1 day, could be up to 2."
49 if (self._num_days(self._today()) - ts) > settings.PASSWORD_RESET_TIMEOUT_DAYS:
50 return False
51  
52 return True

54 def _make_token_with_timestamp(self, user, timestamp):
55 # timestamp is number of days since 2001-1-1. Converted to
56 # base 36, this gives us a 3 digit string until about 2121
57 ts_b36 = int_to_base36(timestamp)
58 hash_string = salted_hmac(
59 self.key_salt,
60 self._make_hash_value(user, timestamp),
61 secret=self.secret,
62 ).hexdigest()[::2] # Limit to 20 characters to shorten the URL.
63 return "%s-%s" % (ts_b36, hash_string)

65 def _make_hash_value(self, user, timestamp):
66 """
67 Hash the user's primary key and some user state that's sure to change
68 after a password reset to produce a token that invalidated when it's
69 used:
70 1. The password field will change upon a password reset (even if the
71 same password is chosen, due to password salting).
72 2. The last_login field will usually be updated very shortly after
73 a password reset.
74 Failing those things, settings.PASSWORD_RESET_TIMEOUT_DAYS eventually
75 invalidates the token.
76  
77 Running this data through salted_hmac() prevents password cracking
78 attempts using the reset token, provided the secret isn't compromised.
79 """
