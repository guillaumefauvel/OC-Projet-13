159 def spatial_version(self):
160 """Determine the version of the PostGIS library."""
161 # Trying to get the PostGIS version because the function
162 # signatures will depend on the version used. The cost
163 # here is a database query to determine the version, which
164 # can be mitigated by setting `POSTGIS_VERSION` with a 3-tuple
165 # comprising user-supplied values for the major, minor, and
166 # subminor revision of PostGIS.
167 if hasattr(settings, 'POSTGIS_VERSION'):
168 version = settings.POSTGIS_VERSION
169 else:
170 # Run a basic query to check the status of the connection so we're
171 # sure we only raise the error below if the problem comes from
172 # PostGIS and not from PostgreSQL itself (see #24862).
173 self._get_postgis_func('version')
174  
175 try:
176 vtup = self.postgis_version_tuple()
177 except ProgrammingError:
178 raise ImproperlyConfigured(
179 'Cannot determine PostGIS version for database "%s" '
180 'using command "SELECT postgis_lib_version()". '
181 'GeoDjango requires at least PostGIS version 2.2. '
182 'Was the database created from a spatial database '
183 'template?' % self.connection.settings_dict['NAME']
184 )
185 version = vtup[1:]
186 return version

188 def convert_extent(self, box):
189 """
190 Return a 4-tuple extent for the `Extent` aggregate by converting
191 the bounding box text returned by PostGIS (`box` argument), for
192 example: "BOX(-90.0 30.0, -85.0 40.0)".
193 """
194 if box is None:
195 return None
196 ll, ur = box[4:-1].split(',')
197 xmin, ymin = map(float, ll.split())
198 xmax, ymax = map(float, ur.split())
199 return (xmin, ymin, xmax, ymax)

201 def convert_extent3d(self, box3d):
202 """
203 Return a 6-tuple extent for the `Extent3D` aggregate by converting

207 if box3d is None:

334 def proj_version_tuple(self):
335 """
336 Return the version of PROJ.4 used by PostGIS as a tuple of the
337 major, minor, and subminor release numbers.
338 """
339 proj_regex = re.compile(r'(\d+)\.(\d+)\.(\d+)')
340 proj_ver_str = self.postgis_proj_version()
341 m = proj_regex.search(proj_ver_str)
342 if m:
343 return tuple(map(int, [m.group(1), m.group(2), m.group(3)]))
344 else:
345 raise Exception('Could not determine PROJ.4 version from PostGIS.')
