181 def build_absolute_uri(self, location=None):
182 """
183 Build an absolute URI from the location and the variables available in
184 this request. If no ``location`` is specified, build the absolute URI
185 using request.get_full_path(). If the location is absolute, convert it
186 to an RFC 3987 compliant URI and return it. If location is relative or
187 is scheme-relative (i.e., ``//example.com/``), urljoin() it to a base
188 URL constructed from the request variables.
189 """
190 if location is None:
191 # Make it an absolute url (but schemeless and domainless) for the
192 # edge case that the path starts with '//'.
193 location = '//%s' % self.get_full_path()
194 bits = urlsplit(location)
195 if not (bits.scheme and bits.netloc):
196 # Handle the simple, most common case. If the location is absolute
197 # and a scheme or host (netloc) isn't provided, skip an expensive
198 # urljoin() as long as no path segments are '.' or '..'.
199 if (bits.path.startswith('/') and not bits.scheme and not bits.netloc and
200 '/./' not in bits.path and '/../' not in bits.path):
201 # If location starts with '//' but has no netloc, reuse the
202 # schema and netloc from the current request. Strip the double
203 # slashes and continue as if it wasn't specified.
204 if location.startswith('//'):
205 location = location[2:]
206 location = self._current_scheme_host + location
207 else:
208 # Join the constructed URL with the provided location, which
209 # allows the provided location to apply query strings to the
210 # base path.
211 location = urljoin(self._current_scheme_host + self.path, location)
212 return iri_to_uri(location)

527 def urlencode(self, safe=None):
528 """
529 Return an encoded string of all query string arguments.
530  
531 `safe` specifies characters which don't require quoting, for example::
532  
533 >>> q = QueryDict(mutable=True)
534 >>> q['next'] = '/a&b/'
535 >>> q.urlencode()
536 'next=%2Fa%26b%2F'
537 >>> q.urlencode(safe='/')
538 'next=/a%26b/'
539 """
540 output = []
541 if safe:
542 safe = safe.encode(self.encoding)
543  
544 def encode(k, v):
545 return '%s=%s' % ((quote(k, safe), quote(v, safe)))
546 else:
547 def encode(k, v):
548 return urlencode({k: v})
549 for k, list_ in self.lists():
550 output.extend(
551 encode(k.encode(self.encoding), str(v).encode(self.encoding))
552 for v in list_
553 )

574 def split_domain_port(host):
575 """
576 Return a (domain, port) tuple from a given host.
577  
578 Returned domain is lowercased. If the host is invalid, the domain will be
579 empty.
580 """
581 host = host.lower()
582  
583 if not host_validation_re.match(host):
584 return '', ''
585  
586 if host[-1] == ']':
587 # It's an IPv6 address without a port.
588 return host, ''
589 bits = host.rsplit(':', 1)
590 domain, port = bits if len(bits) == 2 else (bits[0], '')
591 # Remove a trailing dot (if present) from the domain.
592 domain = domain[:-1] if domain.endswith('.') else domain
593 return domain, port

596 def validate_host(host, allowed_hosts):
597 """
598 Validate the given host for this site.
599  
600 Check that the host looks valid and matches a host or host pattern in the
601 given list of ``allowed_hosts``. Any pattern beginning with a period
602 matches a domain and all its subdomains (e.g. ``.example.com`` matches
603 ``example.com`` and any subdomain), ``*`` matches anything, and anything
604 else must match exactly.
605  
606 Note: This function assumes that the given host is lowercased and has
607 already had the port, if any, stripped off.
608  
609 Return ``True`` for a valid host, ``False`` otherwise.
610 """
611 return any(pattern == '*' or is_same_domain(host, pattern) for pattern in allowed_hosts)
