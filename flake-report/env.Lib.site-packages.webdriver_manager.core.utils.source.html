105 def linux_browser_apps_to_cmd(*apps: str) -> str:
106 """Create 'browser --version' command from browser app names.
107  
108 Result command example:
109 chromium --version || chromium-browser --version
110 """
111 ignore_errors_cmd_part = " 2>/dev/null" if os.getenv(

116 def windows_browser_apps_to_cmd(*apps: str) -> str:
117 """Create analogue of browser --version command for windows."""
118 powershell = determine_powershell()
119  
120 first_hit_template = """$tmp = {expression}; if ($tmp) {{echo $tmp; Exit;}};"""
121 script = "$ErrorActionPreference='silentlycontinue'; " + " ".join(
122 first_hit_template.format(expression=e) for e in apps
123 )
124  
125 return f'{powershell} -NoProfile "{script}"'

233 def read_version_from_cmd(cmd, pattern):
234 with subprocess.Popen(
235 cmd,
236 stdout=subprocess.PIPE,
237 stderr=subprocess.DEVNULL,
238 stdin=subprocess.DEVNULL,
239 shell=True,
240 ) as stream:
241 stdout = stream.communicate()[0].decode()
242 version = re.search(pattern, stdout)
243 version = version.group(0) if version else None
244 return version

247 def determine_powershell():
248 """Returns "True" if runs in Powershell and "False" if another console."""
249 cmd = "(dir 2>&1 *`|echo CMD);&<# rem #>echo powershell"
250 with subprocess.Popen(
251 cmd,
252 stdout=subprocess.PIPE,
253 stderr=subprocess.DEVNULL,
254 stdin=subprocess.DEVNULL,
255 shell=True,
256 ) as stream:
257 stdout = stream.communicate()[0].decode()
258 return "" if stdout == "powershell" else "powershell"
