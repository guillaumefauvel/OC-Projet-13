31 def __init__(self, executable_path=DEFAULT_EXECUTABLE_PATH,
32 port=0, options=None, desired_capabilities=None,
33 service_log_path=None, keep_alive=False):
34 """
35 Creates a new instance of the WebKitGTK driver.
36  
37 Starts the service and then creates new instance of WebKitGTK Driver.
38  
39 :Args:
  • E501 Line too long (118 > 99 characters)
40 - executable_path : path to the executable. If the default is used it assumes the executable is in the $PATH.
41 - port : port you would like the service to run, if left as 0, a free port will be found.
42 - options : an instance of WebKitGTKOptions
43 - desired_capabilities : Dictionary object with desired capabilities
44 - service_log_path : Path to write service stdout and stderr output.
45 - keep_alive : Whether to configure RemoteConnection to use HTTP keep-alive.
46 """
47 if not options:
48 if not desired_capabilities:
49 desired_capabilities = Options().to_capabilities()
50 else:
51 capabilities = options.to_capabilities()
52 if desired_capabilities:
53 capabilities.update(desired_capabilities)
54 desired_capabilities = capabilities
55  
56 self.service = Service(executable_path, port=port, log_path=service_log_path)
57 self.service.start()
58  
59 RemoteWebDriver.__init__(
60 self,
61 command_executor=self.service.service_url,
62 desired_capabilities=desired_capabilities,
63 keep_alive=keep_alive)
64 self._is_remote = False

66 def quit(self):
67 """
68 Closes the browser and shuts down the WebKitGTKDriver executable
69 that is started when starting the WebKitGTKDriver
70 """
71 try:
72 RemoteWebDriver.quit(self)
73 except http_client.BadStatusLine:
74 pass
75 finally:
76 self.service.stop()
