219 def ascii_escaped(val: Union[bytes, str]) -> str:
220 r"""If val is pure ASCII, return it as an str, otherwise, escape
221 bytes objects into a sequence of escaped bytes:
222  
223 b'\xc3\xb4\xc5\xd6' -> r'\xc3\xb4\xc5\xd6'
224  
225 and escapes unicode objects into a sequence of escaped unicode

227  
228 r'4\nV\U00043efa\x0eMXWB\x1e\u3028\u15fd\xcd\U0007d944'
229  
230 Note:
231 The obvious "v.decode('unicode-escape')" will return

373 # Perform exhaustiveness checking.
374 #
375 # Consider this example:
376 #
377 # MyUnion = Union[int, str]
378 #
379 # def handle(x: MyUnion) -> int {
380 # if isinstance(x, int):
381 # return 1
382 # elif isinstance(x, str):
383 # return 2
384 # else:
385 # raise Exception('unreachable')
386 #
387 # Now suppose we add a new variant:
388 #
389 # MyUnion = Union[int, str, bytes]
390 #
391 # After doing this, we must remember ourselves to go and update the handle
392 # function to handle the new variant.
393 #
394 # With `assert_never` we can do better:
395 #
396 # // raise Exception('unreachable')
397 # return assert_never(x)
398 #
399 # Now, if we forget to handle the new variant, the type-checker will emit a
400 # compile-time error, instead of the runtime error we would have gotten
401 # previously.
402 #
403 # This also work for Enums (if you use `is` to compare) and Literals.
404 def assert_never(value: "NoReturn") -> "NoReturn":
405 assert False, f"Unhandled value: {value} ({type(value).__name__})"
