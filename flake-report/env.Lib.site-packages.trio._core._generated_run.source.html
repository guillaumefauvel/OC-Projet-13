105 def spawn_system_task(async_fn, *args, name=None, context=None):
106 """Spawn a "system" task.
107  
108 System tasks have a few differences from regular tasks:
109  
110 * They don't need an explicit nursery; instead they go into the
111 internal "system nursery".
112  
113 * If a system task raises an exception, then it's converted into a
114 :exc:`~trio.TrioInternalError` and *all* tasks are cancelled. If you
115 write a system task, you should be careful to make sure it doesn't
116 crash.
117  
118 * System tasks are automatically cancelled when the main task exits.
119  
120 * By default, system tasks have :exc:`KeyboardInterrupt` protection
121 *enabled*. If you want your task to be interruptible by control-C,
122 then you need to use :func:`disable_ki_protection` explicitly (and
123 come up with some plan for what to do with a
124 :exc:`KeyboardInterrupt`, given that system tasks aren't allowed to
125 raise exceptions).
126  
127 * System tasks do not inherit context variables from their creator.
128  
129 Towards the end of a call to :meth:`trio.run`, after the main
130 task and all system tasks have exited, the system nursery
131 becomes closed. At this point, new calls to
132 :func:`spawn_system_task` will raise ``RuntimeError("Nursery
133 is closed to new arrivals")`` instead of creating a system
134 task. It's possible to encounter this state either in
135 a ``finally`` block in an async generator, or in a callback
136 passed to :meth:`TrioToken.run_sync_soon` at the right moment.
137  
138 Args:
139 async_fn: An async callable.
140 args: Positional arguments for ``async_fn``. If you want to pass
141 keyword arguments, use :func:`functools.partial`.
142 name: The name for this task. Only used for debugging/introspection

144 :func:`spawn_system_task` will try to make it one. A common use

147 make debugging easier.

176 async def wait_all_tasks_blocked(cushion=0.0):
177 """Block until there are no runnable tasks.
178  
179 This is useful in testing code when you want to give other tasks a
180 chance to "settle down". The calling task is blocked, and doesn't wake
181 up until all other tasks are also blocked for at least ``cushion``
182 seconds. (Setting a non-zero ``cushion`` is intended to handle cases
183 like two tasks talking to each other over a local socket, where we
184 want to ignore the potential brief moment between a send and receive
185 when all tasks are blocked.)
186  
187 Note that ``cushion`` is measured in *real* time, not the Trio clock
188 time.
189  
190 If there are multiple tasks blocked in :func:`wait_all_tasks_blocked`,
191 then the one with the shortest ``cushion`` is the one woken (and
192 this task becoming unblocked resets the timers for the remaining
193 tasks). If there are multiple tasks that have exactly the same
194 ``cushion``, then all are woken.
195  
196 You should also consider :class:`trio.testing.Sequencer`, which
197 provides a more explicit way to control execution ordering within a
198 test, and will often produce more readable tests.
199  
200 Example:
201 Here's an example of one way to test that Trio's locks are fair: we
202 take the lock in the parent, start a child, wait for the child to be
203 blocked waiting for the lock (!), and then check that we can't
204 release and immediately re-acquire the lock::
205  
206 async def lock_taker(lock):
207 await lock.acquire()
208 lock.release()
209  
210 async def test_lock_fairness():
211 lock = trio.Lock()
212 await lock.acquire()
213 async with trio.open_nursery() as nursery:
214 nursery.start_soon(lock_taker, lock)
215 # child hasn't run yet, we have the lock
216 assert lock.locked()
217 assert lock._owner is trio.lowlevel.current_task()
218 await trio.testing.wait_all_tasks_blocked()
219 # now the child has run and is blocked on lock.acquire(), we
220 # still have the lock
221 assert lock.locked()
222 assert lock._owner is trio.lowlevel.current_task()
223 lock.release()
224 try:
225 # The child has a prior claim, so we can't have it
226 lock.acquire_nowait()
227 except trio.WouldBlock:
228 assert lock._owner is not trio.lowlevel.current_task()
229 print("PASS")
230 else:
231 print("FAIL")
232  
233 """
