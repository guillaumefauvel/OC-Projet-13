14 # We have a tree of MultiError's, like:
15 #
16 # MultiError([
17 # ValueError,
18 # MultiError([
19 # KeyError,
20 # ValueError,
21 # ]),
22 # ])
23 #
24 # or similar.
25 #
26 # We want to
27 # 1) apply the filter to each of the leaf exceptions -- each leaf
28 # might stay the same, be replaced (with the original exception
29 # potentially sticking around as __context__ or __cause__), or
30 # disappear altogether.
31 # 2) simplify the resulting tree -- remove empty nodes, and replace
32 # singleton MultiError's with their contents, e.g.:
33 # MultiError([KeyError]) -> KeyError
34 # (This can happen recursively, e.g. if the two ValueErrors above
35 # get caught then we'll just be left with a bare KeyError.)
36 # 3) preserve sensible tracebacks
37 #
38 # It's the tracebacks that are most confusing. As a MultiError
39 # propagates through the stack, it accumulates traceback frames, but
40 # the exceptions inside it don't. Semantically, the traceback for a
41 # leaf exception is the concatenation the tracebacks of all the
42 # exceptions you see when traversing the exception tree from the root
43 # to that leaf. Our correctness invariant is that this concatenated
44 # traceback should be the same before and after.
45 #
46 # The easy way to do that would be to, at the beginning of this
47 # function, "push" all tracebacks down to the leafs, so all the
48 # MultiErrors have __traceback__=None, and all the leafs have complete
49 # tracebacks. But whenever possible, we'd actually prefer to keep
50 # tracebacks as high up in the tree as possible, because this lets us
51 # keep only a single copy of the common parts of these exception's
52 # tracebacks. This is cheaper (in memory + time -- tracebacks are
53 # unpleasantly quadratic-ish to work with, and this might matter if
54 # you have thousands of exceptions, which can happen e.g. after
55 # cancelling a large task pool, and no-one will ever look at their
56 # tracebacks!), and more importantly, factoring out redundant parts of
57 # the tracebacks makes them more readable if/when users do see them.
58 #
59 # So instead our strategy is:
60 # - first go through and construct the new tree, preserving any
61 # unchanged subtrees
62 # - then go through the original tree (!) and push tracebacks down
63 # until either we hit a leaf, or we hit a subtree which was
64 # preserved in the new tree.
65  
66 # This used to also support async handler functions. But that runs into:
67 # https://bugs.python.org/issue29600
68 # which is difficult to fix on our end.

155 """An exception that contains other exceptions; also known as an
156 "inception".
157  
158 It's main use is to represent the situation when multiple child tasks all
159 raise errors "in parallel".
160  
161 Args:
162 exceptions (list): The exceptions
163  
164 Returns:
165 If ``len(exceptions) == 1``, returns that exception. This means that a
166 call to ``MultiError(...)`` is not guaranteed to return a
167 :exc:`MultiError` object!

173 :exc:`BaseException`.

312 # TracebackType has no public constructor, so allocate one the hard way
313 try:
314 raise ValueError
315 except ValueError as exc:
316 new_tb = exc.__traceback__
317 c_new_tb = CTraceback.from_address(id(new_tb))
318  
319 # At the C level, tb_next either pointer to the next traceback or is
320 # NULL. c_void_p and the .tb_next accessor both convert NULL to None,
321 # but we shouldn't DECREF None just because we assigned to a NULL
322 # pointer! Here we know that our new traceback has only 1 frame in it,
323 # so we can assume the tb_next field is NULL.
324 assert c_new_tb.tb_next is None
325 # If tb_next is None, then we want to set c_new_tb.tb_next to NULL,
326 # which it already is, so we're done. Otherwise, we have to actually
327 # do some work:
328 if tb_next is not None:

349 def concat_tb(head, tail):
350 # We have to use an iterative algorithm here, because in the worst case
351 # this might be a RecursionError stack that is by definition too deep to
352 # process by recursion!
353 head_tbs = []
354 pointer = head
355 while pointer is not None:
356 head_tbs.append(pointer)
357 pointer = pointer.tb_next
358 current_head = tail
359 for head_tb in reversed(head_tbs):
360 current_head = copy_tb(head_tb, tb_next=current_head)
361 return current_head
