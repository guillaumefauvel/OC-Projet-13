162 def _get_asn1_time(timestamp):
163 """
164 Retrieve the time value of an ASN1 time object.
165  
166 @param timestamp: An ASN1_GENERALIZEDTIME* (or an object safely castable to
167 that type) from which the time value will be retrieved.
168  
169 @return: The time value from C{timestamp} as a L{bytes} string in a certain
170 format. Or C{None} if the object contains no time value.
171 """
172 string_timestamp = _ffi.cast("ASN1_STRING*", timestamp)
173 if _lib.ASN1_STRING_length(string_timestamp) == 0:
174 return None
175 elif (
176 _lib.ASN1_STRING_type(string_timestamp) == _lib.V_ASN1_GENERALIZEDTIME
177 ):
178 return _ffi.string(_lib.ASN1_STRING_data(string_timestamp))
179 else:
180 generalized_timestamp = _ffi.new("ASN1_GENERALIZEDTIME**")
181 _lib.ASN1_TIME_to_generalizedtime(timestamp, generalized_timestamp)
182 if generalized_timestamp[0] == _ffi.NULL:
183 # This may happen:
184 # - if timestamp was not an ASN1_TIME
185 # - if allocating memory for the ASN1_GENERALIZEDTIME failed
186 # - if a copy of the time data from timestamp cannot be made for
187 # the newly allocated ASN1_GENERALIZEDTIME
188 #
189 # These are difficult to test. cffi enforces the ASN1_TIME type.
190 # Memory allocation failures are a pain to trigger
191 # deterministically.
192 _untested_error("ASN1_TIME_to_generalizedtime")
193 else:
194 string_timestamp = _ffi.cast(
195 "ASN1_STRING*", generalized_timestamp[0]
196 )
197 string_data = _lib.ASN1_STRING_data(string_timestamp)
198 string_result = _ffi.string(string_data)
199 _lib.ASN1_GENERALIZEDTIME_free(generalized_timestamp[0])
200 return string_result

1759 class X509StoreContext(object):
1760 """
1761 An X.509 store context.
1762  
1763 An X.509 store context is used to carry out the actual verification process
1764 of a certificate in a described context. For describing such a context, see
1765 :class:`X509Store`.
1766  
1767 :ivar _store_ctx: The underlying X509_STORE_CTX structure used by this
1768 instance. It is dynamically allocated and automatically garbage
1769 collected.
1770 :ivar _store: See the ``store`` ``__init__`` parameter.
1771 :ivar _cert: See the ``certificate`` ``__init__`` parameter.
1772 :ivar _chain: See the ``chain`` ``__init__`` parameter.
1773 :param X509Store store: The certificates which will be trusted for the
1774 purposes of any verifications.
1775 :param X509 certificate: The certificate to be verified.
1776 :param chain: List of untrusted certificates that may be used for building
1777 the certificate chain. May be ``None``.
1778 :type chain: :class:`list` of :class:`X509`
1779 """
1780  
1781 def __init__(self, store, certificate, chain=None):
1782 store_ctx = _lib.X509_STORE_CTX_new()
1783 self._store_ctx = _ffi.gc(store_ctx, _lib.X509_STORE_CTX_free)
1784 self._store = store
1785 self._cert = certificate
1786 self._chain = self._build_certificate_stack(chain)
1787 # Make the store context available for use after instantiating this
1788 # class by initializing it now. Per testing, subsequent calls to
1789 # :meth:`_init` have no adverse affect.
1790 self._init()
1791  
1792 @staticmethod
1793 def _build_certificate_stack(certificates):
1794 def cleanup(s):
1795 # Equivalent to sk_X509_pop_free, but we don't
1796 # currently have a CFFI binding for that available
1797 for i in range(_lib.sk_X509_num(s)):
1798 x = _lib.sk_X509_value(s, i)
1799 _lib.X509_free(x)
1800 _lib.sk_X509_free(s)
1801  
1802 if certificates is None or len(certificates) == 0:
1803 return _ffi.NULL
1804  
1805 stack = _lib.sk_X509_new_null()
1806 _openssl_assert(stack != _ffi.NULL)
1807 stack = _ffi.gc(stack, cleanup)
1808  
1809 for cert in certificates:
1810 if not isinstance(cert, X509):
1811 raise TypeError("One of the elements is not an X509 instance")
1812  
1813 _openssl_assert(_lib.X509_up_ref(cert._x509) > 0)
1814 if _lib.sk_X509_push(stack, cert._x509) <= 0:
1815 _lib.X509_free(cert._x509)
1816 _raise_current_error()
1817  
1818 return stack
