31 def __init__(self, firefox_path=None, log_file=None):
32 """
33 Creates a new instance of Firefox binary.
34  
35 :Args:
  • E501 Line too long (118 > 99 characters)
36 - firefox_path - Path to the Firefox executable. By default, it will be detected from the standard locations.
  • E501 Line too long (100 > 99 characters)
37 - log_file - A file object to redirect the firefox process output to. It can be sys.stdout.
38 Please note that with parallel run the output won't be synchronous.
39 By default, it will be redirected to /dev/null.
40 """
41 self._start_cmd = firefox_path
42 # We used to default to subprocess.PIPE instead of /dev/null, but after
43 # a while the pipe would fill up and Firefox would freeze.
44 self._log_file = log_file or open(os.devnull, "wb")
45 self.command_line = None
46 self.platform = system().lower()
47 if not self._start_cmd:
48 self._start_cmd = self._get_firefox_start_cmd()
49 if not self._start_cmd.strip():
50 raise WebDriverException(
51 "Failed to find firefox binary. You can set it by specifying "
52 "the path to 'firefox_binary':\n\nfrom "
53 "selenium.webdriver.firefox.firefox_binary import "
54 "FirefoxBinary\n\nbinary = "
55 "FirefoxBinary('/path/to/binary')\ndriver = "
56 "webdriver.Firefox(firefox_binary=binary)")
57 # Rather than modifying the environment of the calling Python process
58 # copy it and modify as needed.
59 self._firefox_env = os.environ.copy()
60 self._firefox_env["MOZ_CRASHREPORTER_DISABLE"] = "1"
61 self._firefox_env["MOZ_NO_REMOTE"] = "1"
62 self._firefox_env["NO_EM_RESTART"] = "1"

67 def launch_browser(self, profile, timeout=30):
68 """Launches the browser for the given profile name.
69 It is assumed the profile already exists.

85 def _start_from_profile_path(self, path):
86 self._firefox_env["XRE_PROFILE_PATH"] = path
87  
88 if self.platform == 'linux':
89 self._modify_link_library_path()
90 command = [self._start_cmd, "-foreground"]
91 if self.command_line:
92 for cli in self.command_line:
93 command.append(cli)
94 self.process = Popen(
95 command, stdout=self._log_file, stderr=STDOUT,
96 env=self._firefox_env)

98 def _wait_until_connectable(self, timeout=30):
99 """Blocks until the extension is connectable in the firefox."""
100 count = 0
101 while not utils.is_connectable(self.profile.port):
102 if self.process.poll():
103 # Browser has exited
104 raise WebDriverException(
105 "The browser appears to have exited "
106 "before we could connect. If you specified a log_file in "
107 "the FirefoxBinary constructor, check it for details.")
108 if count >= timeout:
109 self.kill()
110 raise WebDriverException(
111 "Can't load the profile. Possible firefox version mismatch. "
112 "You must use GeckoDriver instead for Firefox 48+. Profile "
113 "Dir: %s If you specified a log_file in the "
114 "FirefoxBinary constructor, check it for details."
115 % (self.profile.path))
116 count += 1
117 time.sleep(1)
118 return True

120 def _find_exe_in_registry(self):
121 try:
122 from _winreg import OpenKey, QueryValue, HKEY_LOCAL_MACHINE, HKEY_CURRENT_USER
123 except ImportError:
124 from winreg import OpenKey, QueryValue, HKEY_LOCAL_MACHINE, HKEY_CURRENT_USER
125 import shlex
126 keys = (r"SOFTWARE\Classes\FirefoxHTML\shell\open\command",
127 r"SOFTWARE\Classes\Applications\firefox.exe\shell\open\command")
128 command = ""
129 for path in keys:
130 try:
131 key = OpenKey(HKEY_LOCAL_MACHINE, path)
132 command = QueryValue(key, "")
133 break
134 except OSError:
135 try:
136 key = OpenKey(HKEY_CURRENT_USER, path)
137 command = QueryValue(key, "")
138 break
139 except OSError:
140 pass
141 else:
142 return ""
143  
144 if not command:
145 return ""
146  
147 return shlex.split(command)[0]

149 def _get_firefox_start_cmd(self):
150 """Return the command to start firefox."""
151 start_cmd = ""
152 if self.platform == "darwin": # small darwin due to lower() in self.platform
153 start_cmd = "/Applications/Firefox.app/Contents/MacOS/firefox-bin"
154 # fallback to homebrew installation for mac users
155 if not os.path.exists(start_cmd):
156 start_cmd = os.path.expanduser("~") + start_cmd
157 elif self.platform == "windows": # same
158 start_cmd = (self._find_exe_in_registry() or self._default_windows_location())
159 elif self.platform == 'java' and os._name == 'nt':
160 start_cmd = self._default_windows_location()
161 else:
162 for ffname in ["firefox", "iceweasel"]:
163 start_cmd = self.which(ffname)
164 if start_cmd:
165 break
166 else:
167 # couldn't find firefox on the system path
168 raise RuntimeError(
169 "Could not find firefox in your system PATH."
170 " Please specify the firefox binary location or install firefox")
171 return start_cmd

173 def _default_windows_location(self):
174 program_files = [os.getenv("PROGRAMFILES", r"C:\Program Files"),
175 os.getenv("PROGRAMFILES(X86)", r"C:\Program Files (x86)")]
176 for path in program_files:
177 binary_path = os.path.join(path, r"Mozilla Firefox\firefox.exe")
178 if os.access(binary_path, os.X_OK):
179 return binary_path
180 return ""

182 def _modify_link_library_path(self):
183 existing_ld_lib_path = os.environ.get('LD_LIBRARY_PATH', '')
184  
185 new_ld_lib_path = self._extract_and_check(
186 self.profile, self.NO_FOCUS_LIBRARY_NAME, "x86", "amd64")
187  
188 new_ld_lib_path += existing_ld_lib_path
189  
190 self._firefox_env["LD_LIBRARY_PATH"] = new_ld_lib_path
191 self._firefox_env['LD_PRELOAD'] = self.NO_FOCUS_LIBRARY_NAME

193 def _extract_and_check(self, profile, no_focus_so_name, x86, amd64):
194  
195 paths = [x86, amd64]
196 built_path = ""
197 for path in paths:
198 library_path = os.path.join(profile.path, path)
199 if not os.path.exists(library_path):
200 os.makedirs(library_path)
201 import shutil
202 shutil.copy(os.path.join(
203 os.path.dirname(__file__),
204 path,
205 self.NO_FOCUS_LIBRARY_NAME),
206 library_path)
207 built_path += library_path + ":"
208  
209 return built_path

211 def which(self, fname):
212 """Returns the fully qualified path by searching Path of the given
213 name"""
214 for pe in os.environ['PATH'].split(os.pathsep):
215 checkname = os.path.join(pe, fname)
216 if os.access(checkname, os.X_OK) and not os.path.isdir(checkname):
217 return checkname
218 return None
