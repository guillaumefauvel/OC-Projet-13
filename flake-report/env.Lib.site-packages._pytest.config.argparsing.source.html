473 class DropShorterLongHelpFormatter(argparse.HelpFormatter):
474 """Shorten help for long options that differ only in extra hyphens.
475  
476 - Collapse **long** options that are the same except for extra hyphens.
477 - Shortcut if there are only two options and one of them is a short one.
478 - Cache result on the action object as this is called at least 2 times.
479 """

481 def __init__(self, *args: Any, **kwargs: Any) -> None:
482 # Use more accurate terminal width.
483 if "width" not in kwargs:
484 kwargs["width"] = _pytest._io.get_terminal_width()

487 def _format_action_invocation(self, action: argparse.Action) -> str:
488 orgstr = super()._format_action_invocation(action)
489 if orgstr and orgstr[0] != "-": # only optional arguments
490 return orgstr
491 res: Optional[str] = getattr(action, "_formatted_action_invocation", None)
492 if res:
493 return res
494 options = orgstr.split(", ")
495 if len(options) == 2 and (len(options[0]) == 2 or len(options[1]) == 2):
496 # a shortcut for '-h, --help' or '--abc', '-a'
497 action._formatted_action_invocation = orgstr # type: ignore
498 return orgstr
499 return_list = []
500 short_long: Dict[str, str] = {}
501 for option in options:
502 if len(option) == 2 or option[2] == " ":
503 continue
504 if not option.startswith("--"):
505 raise ArgumentError(
506 'long optional argument without "--": [%s]' % (option), option
507 )
508 xxoption = option[2:]
509 shortened = xxoption.replace("-", "")
510 if shortened not in short_long or len(short_long[shortened]) < len(
511 xxoption
512 ):
513 short_long[shortened] = xxoption
514 # now short_long has been filled out to the longest with dashes
515 # **and** we keep the right option ordering from add_argument
516 for option in options:
517 if len(option) == 2 or option[2] == " ":
518 return_list.append(option)
519 if option[2:] == short_long.get(option.replace("-", "")):
520 return_list.append(option.replace(" ", "=", 1))
521 formatted_action_invocation = ", ".join(return_list)
522 action._formatted_action_invocation = formatted_action_invocation # type: ignore
523 return formatted_action_invocation
