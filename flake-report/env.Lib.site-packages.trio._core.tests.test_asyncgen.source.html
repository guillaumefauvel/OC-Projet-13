232 async def step_outside_async_context(aiter):
233 # abort_fns run outside of task context, at least if they're
234 # triggered by a deadline expiry rather than a direct
235 # cancellation. Thus, an asyncgen first iterated inside one
236 # will appear non-Trio, and since no other hooks were installed,
237 # will use the last-ditch fallback handling (that tries to mimic
238 # CPython's behavior with no hooks).
239 #
240 # NB: the strangeness with aiter being an attribute of abort_fn is
241 # to make it as easy as possible to ensure we don't hang onto a
242 # reference to aiter inside the guts of the run loop.
243 def abort_fn(_):
244 with pytest.raises(StopIteration, match="42"):
245 abort_fn.aiter.asend(None).send(None)
246 del abort_fn.aiter
247 return _core.Abort.SUCCEEDED
248  
249 abort_fn.aiter = aiter
250  
251 async with _core.open_nursery() as nursery:
252 nursery.start_soon(_core.wait_task_rescheduled, abort_fn)
253 await _core.wait_all_tasks_blocked()
254 nursery.cancel_scope.deadline = _core.current_time()
