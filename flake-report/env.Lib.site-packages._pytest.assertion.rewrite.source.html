287 def _write_pyc_fp(
288 fp: IO[bytes], source_stat: os.stat_result, co: types.CodeType
289 ) -> None:
290 # Technically, we don't have to have the same pyc format as
291 # (C)Python, since these "pycs" should never be seen by builtin
292 # import. However, there's little reason to deviate.
293 fp.write(importlib.util.MAGIC_NUMBER)
294 # https://www.python.org/dev/peps/pep-0552/
295 flags = b"\x00\x00\x00\x00"

365 def _read_pyc(
366 source: Path, pyc: Path, trace: Callable[[str], None] = lambda x: None
367 ) -> Optional[types.CodeType]:
368 """Possibly read a pytest pyc containing rewritten code.
369  
370 Return rewritten code if successful or None if not.
371 """
372 try:
373 fp = open(pyc, "rb")
374 except OSError:
375 return None
376 with fp:
377 try:
378 stat_result = os.stat(source)
379 mtime = int(stat_result.st_mtime)
380 size = stat_result.st_size
381 data = fp.read(16)
382 except OSError as e:
383 trace(f"_read_pyc({source}): OSError {e}")
384 return None
385 # Check for invalid or out of date pyc file.
386 if len(data) != (16):
387 trace("_read_pyc(%s): invalid pyc (too short)" % source)
388 return None
389 if data[:4] != importlib.util.MAGIC_NUMBER:
390 trace("_read_pyc(%s): invalid pyc (bad magic number)" % source)
391 return None
392 if data[4:8] != b"\x00\x00\x00\x00":
393 trace("_read_pyc(%s): invalid pyc (unsupported flags)" % source)
394 return None
395 mtime_data = data[8:12]
396 if int.from_bytes(mtime_data, "little") != mtime & 0xFFFFFFFF:
397 trace("_read_pyc(%s): out of date" % source)
398 return None
399 size_data = data[12:16]
400 if int.from_bytes(size_data, "little") != size & 0xFFFFFFFF:
401 trace("_read_pyc(%s): invalid pyc (incorrect size)" % source)
402 return None
403 try:
404 co = marshal.load(fp)
405 except Exception as e:
406 trace(f"_read_pyc({source}): marshal.load error {e}")
407 return None
408 if not isinstance(co, types.CodeType):
409 trace("_read_pyc(%s): not a code object" % source)
410 return None
411 return co

607 class AssertionRewriter(ast.NodeVisitor):
608 """Assertion rewriting implementation.
609  
610 The main entrypoint is to call .run() with an ast.Module instance,
611 this will then find all the assert statements and rewrite them to
612 provide intermediate values and a detailed assertion error. See
613 http://pybites.blogspot.be/2011/07/behind-scenes-of-pytests-new-assertion.html
614 for an overview of how this works.
615  
616 The entry point here is .run() which will iterate over all the
617 statements in an ast.Module and for each ast.Assert statement it
618 finds call .visit() with it. Then .visit_Assert() takes over and
619 is responsible for creating new ast statements to replace the
620 original assert statement: it rewrites the test of an assertion
621 to provide intermediate values and replace it with an if statement
622 which raises an assertion error with a detailed explanation in
623 case the expression is false and calls pytest_assertion_pass hook
624 if expression is true.
625  
626 For this .visit_Assert() uses the visitor pattern to visit all the
627 AST nodes of the ast.Assert.test field, each visit call returning
628 an AST node and the corresponding explanation string. During this
629 state is kept in several instance attributes:
630  
631 :statements: All the AST statements which will replace the assert
632 statement.
633  
634 :variables: This is populated by .variable() with each variable
635 used by the statements so that they can all be set to None at
636 the end of the statements.
637  
638 :variable_counter: Counter to create new unique variables needed
639 by statements. Variables are created using .variable() and
640 have the form of "@py_assert0".
641  
642 :expl_stmts: The AST statements which will be executed to get
643 data from the assertion. This is the code which will construct
644 the detailed assertion message that is used in the AssertionError
645 or for the pytest_assertion_pass hook.
646  
647 :explanation_specifiers: A dict filled by .explanation_param()
648 with %-formatting placeholders and their corresponding
649 expressions to use in the building of an assertion message.
650 This is used by .pop_format_context() to build a message.
651  
652 :stack: A stack of the explanation_specifiers dicts maintained by
653 .push_format_context() and .pop_format_context() which allows
654 to build another %-formatted string while already building one.
655  
656 This state is reset on every new assert statement visited and used
657 by the other visitors.
658 """

674 def run(self, mod: ast.Module) -> None:
675 """Find all assert statements in *mod* and rewrite them."""
676 if not mod.body:
677 # Nothing to do.
678 return
679  
680 # We'll insert some special imports at the top of the module, but after any
681 # docstrings and __future__ imports, so first figure out where that is.
682 doc = getattr(mod, "docstring", None)
683 expect_docstring = doc is None
684 if doc is not None and self.is_rewrite_disabled(doc):
685 return
686 pos = 0
687 lineno = 1
688 for item in mod.body:
689 if (
690 expect_docstring
691 and isinstance(item, ast.Expr)
692 and isinstance(item.value, ast.Str)
693 ):
694 doc = item.value.s
695 if self.is_rewrite_disabled(doc):
696 return
697 expect_docstring = False
698 elif (
699 isinstance(item, ast.ImportFrom)
700 and item.level == 0
701 and item.module == "__future__"
702 ):
703 pass
704 else:
705 break
706 pos += 1
707 # Special case: for a decorated function, set the lineno to that of the
708 # first decorator, not the `def`. Issue #4984.
709 if isinstance(item, ast.FunctionDef) and item.decorator_list:
710 lineno = item.decorator_list[0].lineno
711 else:
712 lineno = item.lineno
713 # Now actually insert the special imports.
714 if sys.version_info >= (3, 10):
715 aliases = [
716 ast.alias("builtins", "@py_builtins", lineno=lineno, col_offset=0),
717 ast.alias(
718 "_pytest.assertion.rewrite",
719 "@pytest_ar",
720 lineno=lineno,
721 col_offset=0,
722 ),
723 ]
724 else:
725 aliases = [
726 ast.alias("builtins", "@py_builtins"),
727 ast.alias("_pytest.assertion.rewrite", "@pytest_ar"),
728 ]
729 imports = [
730 ast.Import([alias], lineno=lineno, col_offset=0) for alias in aliases
731 ]
732 mod.body[pos:pos] = imports
733  
734 # Collect asserts.
735 nodes: List[ast.AST] = [mod]
736 while nodes:
737 node = nodes.pop()
738 for name, field in ast.iter_fields(node):
739 if isinstance(field, list):
740 new: List[ast.AST] = []
741 for i, child in enumerate(field):
742 if isinstance(child, ast.Assert):
743 # Transform assert.
744 new.extend(self.visit(child))
745 else:
746 new.append(child)
747 if isinstance(child, ast.AST):
748 nodes.append(child)
749 setattr(node, name, new)
750 elif (
751 isinstance(field, ast.AST)
752 # Don't recurse into expressions as they can't contain
753 # asserts.
754 and not isinstance(field, ast.expr)
755 ):
756 nodes.append(field)

759 def is_rewrite_disabled(docstring: str) -> bool:
760 return "PYTEST_DONT_REWRITE" in docstring

764 # Use a character invalid in python identifiers to avoid clashing.

766 self.variables.append(name)

790 def explanation_param(self, expr: ast.expr) -> str:
791 """Return a new named %-formatting placeholder for expr.
792  
793 This creates a %-formatting placeholder for expr in the
794 current formatting context, e.g. ``%(py0)s``. The placeholder
795 and expr are placed in the current format context so that it
796 can be used on the next call to .pop_format_context().
797 """
798 specifier = "py" + str(next(self.variable_counter))
799 self.explanation_specifiers[specifier] = expr
800 return "%(" + specifier + ")s"
801  
802 def push_format_context(self) -> None:
803 """Create a new formatting context.
804  
805 The format context is used for when an explanation wants to
806 have a variable value formatted in the assertion message. In
807 this case the value required can be added using
808 .explanation_param(). Finally .pop_format_context() is used
809 to format a string of %-formatted values as added by
810 .explanation_param().
811 """
812 self.explanation_specifiers: Dict[str, ast.expr] = {}
813 self.stack.append(self.explanation_specifiers)
814  
815 def pop_format_context(self, expl_expr: ast.expr) -> ast.Name:
816 """Format the %-formatted string with current format context.
817  
818 The expl_expr should be an str ast.expr instance constructed from
819 the %-placeholders created by .explanation_param(). This will
820 add the required code to format said string to .expl_stmts and
821 return the ast.Name instance of the formatted string.
822 """
823 current = self.stack.pop()
824 if self.stack:
825 self.explanation_specifiers = self.stack[-1]
826 keys = [ast.Str(key) for key in current.keys()]
827 format_dict = ast.Dict(keys, list(current.values()))
828 form = ast.BinOp(expl_expr, ast.Mod(), format_dict)
829 name = "@py_format" + str(next(self.variable_counter))
830 if self.enable_assertion_pass_hook:
831 self.format_variables.append(name)
832 self.expl_stmts.append(ast.Assign([ast.Name(name, ast.Store())], form))
833 return ast.Name(name, ast.Load())

835 def generic_visit(self, node: ast.AST) -> Tuple[ast.Name, str]:
836 """Handle expressions we don't have custom code for."""
837 assert isinstance(node, ast.expr)
838 res = self.assign(node)
839 return res, self.explanation_param(self.display(res))

841 def visit_Assert(self, assert_: ast.Assert) -> List[ast.stmt]:
842 """Return the AST statements to replace the ast.Assert instance.
843  
844 This rewrites the test of an assertion to provide
845 intermediate values and replace it with an if statement which
846 raises an assertion error with a detailed explanation in case
847 the expression is false.
848 """
849 if isinstance(assert_.test, ast.Tuple) and len(assert_.test.elts) >= 1:
850 from _pytest.warning_types import PytestAssertRewriteWarning
851 import warnings
852  
853 # TODO: This assert should not be needed.
854 assert self.module_path is not None
855 warnings.warn_explicit(
856 PytestAssertRewriteWarning(
857 "assertion is always true, perhaps remove parentheses?"
858 ),
859 category=None,
860 filename=self.module_path,
861 lineno=assert_.lineno,
862 )
863  
864 self.statements: List[ast.stmt] = []
865 self.variables: List[str] = []
866 self.variable_counter = itertools.count()
867  
868 if self.enable_assertion_pass_hook:
869 self.format_variables: List[str] = []
870  
871 self.stack: List[Dict[str, ast.expr]] = []
872 self.expl_stmts: List[ast.stmt] = []
873 self.push_format_context()
874 # Rewrite assert into a bunch of statements.
875 top_condition, explanation = self.visit(assert_.test)
876  
877 negation = ast.UnaryOp(ast.Not(), top_condition)
878  
879 if self.enable_assertion_pass_hook: # Experimental pytest_assertion_pass hook
880 msg = self.pop_format_context(ast.Str(explanation))
881  
882 # Failed
883 if assert_.msg:
884 assertmsg = self.helper("_format_assertmsg", assert_.msg)
885 gluestr = "\n>assert "
886 else:
887 assertmsg = ast.Str("")
888 gluestr = "assert "
889 err_explanation = ast.BinOp(ast.Str(gluestr), ast.Add(), msg)
890 err_msg = ast.BinOp(assertmsg, ast.Add(), err_explanation)
891 err_name = ast.Name("AssertionError", ast.Load())
892 fmt = self.helper("_format_explanation", err_msg)
893 exc = ast.Call(err_name, [fmt], [])
894 raise_ = ast.Raise(exc, None)
895 statements_fail = []
896 statements_fail.extend(self.expl_stmts)
897 statements_fail.append(raise_)
898  
899 # Passed
900 fmt_pass = self.helper("_format_explanation", msg)
901 orig = _get_assertion_exprs(self.source)[assert_.lineno]
902 hook_call_pass = ast.Expr(
903 self.helper(
904 "_call_assertion_pass",
905 ast.Num(assert_.lineno),
906 ast.Str(orig),
907 fmt_pass,
908 )
909 )
910 # If any hooks implement assert_pass hook
911 hook_impl_test = ast.If(
912 self.helper("_check_if_assertion_pass_impl"),
913 self.expl_stmts + [hook_call_pass],
914 [],
915 )
916 statements_pass = [hook_impl_test]
917  
918 # Test for assertion condition
919 main_test = ast.If(negation, statements_fail, statements_pass)
920 self.statements.append(main_test)
921 if self.format_variables:
922 variables = [
923 ast.Name(name, ast.Store()) for name in self.format_variables
924 ]
925 clear_format = ast.Assign(variables, ast.NameConstant(None))
926 self.statements.append(clear_format)
927  
928 else: # Original assertion rewriting
929 # Create failure message.
930 body = self.expl_stmts
931 self.statements.append(ast.If(negation, body, []))
932 if assert_.msg:
933 assertmsg = self.helper("_format_assertmsg", assert_.msg)
934 explanation = "\n>assert " + explanation
935 else:
936 assertmsg = ast.Str("")
937 explanation = "assert " + explanation
938 template = ast.BinOp(assertmsg, ast.Add(), ast.Str(explanation))
939 msg = self.pop_format_context(template)
940 fmt = self.helper("_format_explanation", msg)
941 err_name = ast.Name("AssertionError", ast.Load())
942 exc = ast.Call(err_name, [fmt], [])
943 raise_ = ast.Raise(exc, None)
944  
945 body.append(raise_)
946  
947 # Clear temporary variables by setting them to None.
948 if self.variables:
949 variables = [ast.Name(name, ast.Store()) for name in self.variables]
950 clear = ast.Assign(variables, ast.NameConstant(None))
951 self.statements.append(clear)
952 # Fix locations (line numbers/column offsets).
953 for stmt in self.statements:
954 for node in traverse_node(stmt):
955 ast.copy_location(node, assert_)
956 return self.statements

958 def visit_Name(self, name: ast.Name) -> Tuple[ast.Name, str]:
959 # Display the repr of the name if it's a local variable or
960 # _should_repr_global_name() thinks it's acceptable.
961 locs = ast.Call(self.builtin("locals"), [], [])
962 inlocs = ast.Compare(ast.Str(name.id), [ast.In()], [locs])
963 dorepr = self.helper("_should_repr_global_name", name)
964 test = ast.BoolOp(ast.Or(), [inlocs, dorepr])
965 expr = ast.IfExp(test, self.display(name), ast.Str(name.id))
966 return name, self.explanation_param(expr)

968 def visit_BoolOp(self, boolop: ast.BoolOp) -> Tuple[ast.Name, str]:
969 res_var = self.variable()
970 expl_list = self.assign(ast.List([], ast.Load()))
971 app = ast.Attribute(expl_list, "append", ast.Load())
972 is_or = int(isinstance(boolop.op, ast.Or))
973 body = save = self.statements
974 fail_save = self.expl_stmts
975 levels = len(boolop.values) - 1
976 self.push_format_context()
977 # Process each operand, short-circuiting if needed.
978 for i, v in enumerate(boolop.values):
979 if i:
980 fail_inner: List[ast.stmt] = []
981 # cond is set in a prior loop iteration below
982 self.expl_stmts.append(ast.If(cond, fail_inner, [])) # noqa
983 self.expl_stmts = fail_inner
984 self.push_format_context()
985 res, expl = self.visit(v)
986 body.append(ast.Assign([ast.Name(res_var, ast.Store())], res))
987 expl_format = self.pop_format_context(ast.Str(expl))
988 call = ast.Call(app, [expl_format], [])
989 self.expl_stmts.append(ast.Expr(call))
990 if i < levels:
991 cond: ast.expr = res
992 if is_or:
993 cond = ast.UnaryOp(ast.Not(), cond)
994 inner: List[ast.stmt] = []
995 self.statements.append(ast.If(cond, inner, []))
996 self.statements = body = inner
997 self.statements = save
998 self.expl_stmts = fail_save
999 expl_template = self.helper("_format_boolop", expl_list, ast.Num(is_or))
1000 expl = self.pop_format_context(expl_template)
1001 return ast.Name(res_var, ast.Load()), self.explanation_param(expl)

1003 def visit_UnaryOp(self, unary: ast.UnaryOp) -> Tuple[ast.Name, str]:
1004 pattern = UNARY_MAP[unary.op.__class__]

1009 def visit_BinOp(self, binop: ast.BinOp) -> Tuple[ast.Name, str]:
1010 symbol = BINOP_MAP[binop.op.__class__]
1011 left_expr, left_expl = self.visit(binop.left)
1012 right_expr, right_expl = self.visit(binop.right)
1013 explanation = f"({left_expl} {symbol} {right_expl})"
1014 res = self.assign(ast.BinOp(left_expr, binop.op, right_expr))
1015 return res, explanation

1017 def visit_Call(self, call: ast.Call) -> Tuple[ast.Name, str]:
1018 new_func, func_expl = self.visit(call.func)
1019 arg_expls = []
1020 new_args = []
1021 new_kwargs = []
1022 for arg in call.args:
1023 res, expl = self.visit(arg)
1024 arg_expls.append(expl)
1025 new_args.append(res)
1026 for keyword in call.keywords:
1027 res, expl = self.visit(keyword.value)
1028 new_kwargs.append(ast.keyword(keyword.arg, res))
1029 if keyword.arg:
1030 arg_expls.append(keyword.arg + "=" + expl)
1031 else: # **args have `arg` keywords with an .arg of None
1032 arg_expls.append("**" + expl)
1033  
1034 expl = "{}({})".format(func_expl, ", ".join(arg_expls))
1035 new_call = ast.Call(new_func, new_args, new_kwargs)
1036 res = self.assign(new_call)
1037 res_expl = self.explanation_param(self.display(res))
1038 outer_expl = f"{res_expl}\n{{{res_expl} = {expl}\n}}"
1039 return res, outer_expl

1041 def visit_Starred(self, starred: ast.Starred) -> Tuple[ast.Starred, str]:

1043 res, expl = self.visit(starred.value)

1047 def visit_Attribute(self, attr: ast.Attribute) -> Tuple[ast.Name, str]:
1048 if not isinstance(attr.ctx, ast.Load):
1049 return self.generic_visit(attr)
1050 value, value_expl = self.visit(attr.value)
1051 res = self.assign(ast.Attribute(value, attr.attr, ast.Load()))
1052 res_expl = self.explanation_param(self.display(res))
1053 pat = "%s\n{%s = %s.%s\n}"
1054 expl = pat % (res_expl, res_expl, value_expl, attr.attr)
1055 return res, expl

1057 def visit_Compare(self, comp: ast.Compare) -> Tuple[ast.expr, str]:
1058 self.push_format_context()
1059 left_res, left_expl = self.visit(comp.left)
1060 if isinstance(comp.left, (ast.Compare, ast.BoolOp)):
1061 left_expl = f"({left_expl})"
1062 res_variables = [self.variable() for i in range(len(comp.ops))]
1063 load_names = [ast.Name(v, ast.Load()) for v in res_variables]
1064 store_names = [ast.Name(v, ast.Store()) for v in res_variables]
1065 it = zip(range(len(comp.ops)), comp.ops, comp.comparators)
1066 expls = []
1067 syms = []
1068 results = [left_res]
1069 for i, op, next_operand in it:
1070 next_res, next_expl = self.visit(next_operand)
1071 if isinstance(next_operand, (ast.Compare, ast.BoolOp)):
1072 next_expl = f"({next_expl})"
1073 results.append(next_res)
1074 sym = BINOP_MAP[op.__class__]
1075 syms.append(ast.Str(sym))
1076 expl = f"{left_expl} {sym} {next_expl}"
1077 expls.append(ast.Str(expl))
1078 res_expr = ast.Compare(left_res, [op], [next_res])
1079 self.statements.append(ast.Assign([store_names[i]], res_expr))
1080 left_res, left_expl = next_res, next_expl
1081 # Use pytest.assertion.util._reprcompare if that's available.
1082 expl_call = self.helper(
1083 "_call_reprcompare",
1084 ast.Tuple(syms, ast.Load()),
1085 ast.Tuple(load_names, ast.Load()),
1086 ast.Tuple(expls, ast.Load()),
1087 ast.Tuple(results, ast.Load()),
1088 )
1089 if len(comp.ops) > 1:
1090 res: ast.expr = ast.BoolOp(ast.And(), load_names)
1091 else:
1092 res = load_names[0]
1093 return res, self.explanation_param(self.pop_format_context(expl_call))

1096 def try_makedirs(cache_dir: Path) -> bool:
1097 """Attempt to create the given directory and sub-directories exist.
1098  
1099 Returns True if successful or if it already exists.
1100 """
1101 try:
1102 os.makedirs(cache_dir, exist_ok=True)
1103 except (FileNotFoundError, NotADirectoryError, FileExistsError):
1104 # One of the path components was not a directory:
1105 # - we're in a zip file
1106 # - it is a file
1107 return False
1108 except PermissionError:
1109 return False
1110 except OSError as e:
1111 # as of now, EROFS doesn't have an equivalent OSError-subclass
1112 if e.errno == errno.EROFS:
1113 return False
1114 raise
1115 return True

1118 def get_cache_dir(file_path: Path) -> Path:
1119 """Return the cache directory to write .pyc files for the given .py file path."""
1120 if sys.version_info >= (3, 8) and sys.pycache_prefix:
1121 # given:
1122 # prefix = '/tmp/pycs'
1123 # path = '/home/user/proj/test_app.py'
1124 # we want:
1125 # '/tmp/pycs/home/user/proj'
1126 return Path(sys.pycache_prefix) / Path(*file_path.parts[1:-1])
1127 else:
1128 # classic pycache directory
1129 return file_path.parent / "__pycache__"
