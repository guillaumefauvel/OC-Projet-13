1 r"""Evaluate match expressions, as used by `-k` and `-m`.
2  
3 The grammar is:
4  
5 expression: expr? EOF
6 expr: and_expr ('or' and_expr)*
7 and_expr: not_expr ('and' not_expr)*
8 not_expr: 'not' not_expr | '(' expr ')' | ident
9 ident: (\w|:|\+|-|\.|\[|\]|\\|/)+
10  
11 The semantics are:
12  
13 - Empty expression evaluates to False.

73 __slots__ = ("tokens", "current")
74  
75 def __init__(self, input: str) -> None:
76 self.tokens = self.lex(input)
77 self.current = next(self.tokens)

79 def lex(self, input: str) -> Iterator[Token]:
80 pos = 0
81 while pos < len(input):
82 if input[pos] in (" ", "\t"):
83 pos += 1
84 elif input[pos] == "(":
85 yield Token(TokenType.LPAREN, "(", pos)
86 pos += 1
87 elif input[pos] == ")":
88 yield Token(TokenType.RPAREN, ")", pos)
89 pos += 1
90 else:
91 match = re.match(r"(:?\w|:|\+|-|\.|\[|\]|\\|/)+", input[pos:])
92 if match:
93 value = match.group(0)
94 if value == "or":
95 yield Token(TokenType.OR, value, pos)
96 elif value == "and":
97 yield Token(TokenType.AND, value, pos)
98 elif value == "not":
99 yield Token(TokenType.NOT, value, pos)
100 else:
101 yield Token(TokenType.IDENT, value, pos)
102 pos += len(value)
103 else:
104 raise ParseError(
105 pos + 1,
106 f'unexpected character "{input[pos]}"',
107 )
108 yield Token(TokenType.EOF, "", pos)

110 def accept(self, type: TokenType, *, reject: bool = False) -> Optional[Token]:
111 if self.current.type is type:
112 token = self.current
113 if token.type is not TokenType.EOF:
114 self.current = next(self.tokens)
115 return token
116 if reject:
117 self.reject((type,))
118 return None

136 def expression(s: Scanner) -> ast.Expression:
137 if s.accept(TokenType.EOF):
138 ret: ast.expr = ast.NameConstant(False)
139 else:
140 ret = expr(s)
141 s.accept(TokenType.EOF, reject=True)
142 return ast.fix_missing_locations(ast.Expression(ret))

145 def expr(s: Scanner) -> ast.expr:
146 ret = and_expr(s)
147 while s.accept(TokenType.OR):
148 rhs = and_expr(s)
149 ret = ast.BoolOp(ast.Or(), [ret, rhs])
150 return ret
151  
152  
153 def and_expr(s: Scanner) -> ast.expr:
154 ret = not_expr(s)
155 while s.accept(TokenType.AND):
156 rhs = not_expr(s)
157 ret = ast.BoolOp(ast.And(), [ret, rhs])
158 return ret
159  
160  
161 def not_expr(s: Scanner) -> ast.expr:
162 if s.accept(TokenType.NOT):
163 return ast.UnaryOp(ast.Not(), not_expr(s))
164 if s.accept(TokenType.LPAREN):
165 ret = expr(s)
166 s.accept(TokenType.RPAREN, reject=True)
167 return ret
168 ident = s.accept(TokenType.IDENT)
169 if ident:
170 return ast.Name(IDENT_PREFIX + ident.value, ast.Load())
171 s.reject((TokenType.NOT, TokenType.LPAREN, TokenType.IDENT))

203 """Compile a match expression.

215 def evaluate(self, matcher: Callable[[str], bool]) -> bool:
216 """Evaluate the match expression.
217  
218 :param matcher:
219 Given an identifier, should return whether it matches or not.
220 Should be prepared to handle arbitrary strings as input.
221  
222 :returns: Whether the expression matches or not.
223 """
224 ret: bool = eval(self.code, {"__builtins__": {}}, MatcherAdapter(matcher))
225 return ret
