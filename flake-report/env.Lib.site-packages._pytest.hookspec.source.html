206 @hookspec(firstresult=True)
207 def pytest_collection(session: "Session") -> Optional[object]:
208 """Perform the collection phase for the given session.
209  
210 Stops at first non-None result, see :ref:`firstresult`.
211 The return value is not used, but only stops further processing.
212  
213 The default collection phase is this (see individual hooks for full details):
214  
215 1. Starting from ``session`` as the initial collector:
216  
217 1. ``pytest_collectstart(collector)``
218 2. ``report = pytest_make_collect_report(collector)``
  • E501 Line too long (100 > 99 characters)
219 3. ``pytest_exception_interact(collector, call, report)`` if an interactive exception occurred
220 4. For each collected node:
221  
222 1. If an item, ``pytest_itemcollected(item)``
223 2. If a collector, recurse into it.
224  
225 5. ``pytest_collectreport(report)``
226  
227 2. ``pytest_collection_modifyitems(session, config, items)``
228  
229 1. ``pytest_deselected(items)`` for any deselected items (may be called multiple times)
230  
231 3. ``pytest_collection_finish(session)``
232 4. Set ``session.items`` to the list of collected items
233 5. Set ``session.testscollected`` to the number of collected items
234  
235 You can implement this hook to only perform some action before collection,
236 for example the terminal plugin uses it to start displaying the collection
237 counter (and returns `None`).
238  
239 :param pytest.Session session: The pytest session object.
240 """

426 @hookspec(firstresult=True)
427 def pytest_runtest_protocol(
428 item: "Item", nextitem: "Optional[Item]"
429 ) -> Optional[object]:
430 """Perform the runtest protocol for a single test item.
431  
432 The default runtest protocol is this (see individual hooks for full details):
433  
434 - ``pytest_runtest_logstart(nodeid, location)``
435  
436 - Setup phase:
437 - ``call = pytest_runtest_setup(item)`` (wrapped in ``CallInfo(when="setup")``)
438 - ``report = pytest_runtest_makereport(item, call)``
439 - ``pytest_runtest_logreport(report)``
440 - ``pytest_exception_interact(call, report)`` if an interactive exception occurred
441  
442 - Call phase, if the the setup passed and the ``setuponly`` pytest option is not set:
443 - ``call = pytest_runtest_call(item)`` (wrapped in ``CallInfo(when="call")``)
444 - ``report = pytest_runtest_makereport(item, call)``
445 - ``pytest_runtest_logreport(report)``
446 - ``pytest_exception_interact(call, report)`` if an interactive exception occurred
447  
448 - Teardown phase:
  • E501 Line too long (103 > 99 characters)
449 - ``call = pytest_runtest_teardown(item, nextitem)`` (wrapped in ``CallInfo(when="teardown")``)
450 - ``report = pytest_runtest_makereport(item, call)``
451 - ``pytest_runtest_logreport(report)``
452 - ``pytest_exception_interact(call, report)`` if an interactive exception occurred
453  
454 - ``pytest_runtest_logfinish(nodeid, location)``
455  
456 :param item: Test item for which the runtest protocol is performed.
457 :param nextitem: The scheduled-to-be-next test item (or None if this is the end my friend).
458  
459 Stops at first non-None result, see :ref:`firstresult`.
460 The return value is not used, but only stops further processing.
461 """
